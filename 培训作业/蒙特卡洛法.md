#### 作业要求
用计算机仿真布丰实验，计算圆周率

#### 作业分析
- **设定参数**：针的长度 L 和线间的距离 d
- **投掷针**：将针随机地投掷在平行线之间，记录针是否与线相交
- **计算π**：利用实验中针与线相交的次数和总投掷次数来估算 π

#### 实现过程
- python编程实现
```python
import numpy as np
import matplotlib.pyplot as plt

# 参数设置
L = 1.0  # 针的长度
d = 2.0  # 线间距
num_throws = 10000  # 投掷次数

def buffon_needle_experiment(num_throws, L, d):
    intersections = 0  # 与线相交次数
    
    for _ in range(num_throws):
        # 针的中心落点
        x_center = np.random.uniform(0, d / 2)
        # 针与水平线的夹角
        theta = np.random.uniform(0, np.pi / 2)
        
        # 判断是否相交
        if x_center <= (L / 2) * np.sin(theta):
            intersections += 1
    
    # 根据相交次数估算π
    p = intersections / num_throws
    estimated_pi = (2 * L) / (p * d)
    
    return estimated_pi

# 仿真并计算π
estimated_pi = buffon_needle_experiment(num_throws, L, d)
print(f"Estimated value of π after {num_throws} throws: {estimated_pi}")

# 绘制结果
x_vals = np.linspace(0, num_throws, num_throws)
pi_vals = [buffon_needle_experiment(i+1, L, d) for i in range(num_throws)]

plt.figure(figsize=(12, 6))
plt.plot(x_vals, pi_vals, label='Estimated π')
plt.axhline(y=np.pi, color='r', linestyle='-', label='True π')
plt.xlabel('Number of Throws')
plt.ylabel('Estimated π')
plt.legend()
plt.title('Buffon Needle Experiment - Estimating π')
plt.show()
```
