#### 作业分析
- $x^2y^{\prime\prime}+xy^{\prime}+\left(x^2-\left(\frac12\right)^2\right)y=0,\quad y\left(\frac\pi2\right)=2,\quad y^{\prime}\left(\frac\pi2\right)=-\frac2\pi$
- 精确解为$y=\sin(x)\sqrt{\frac{2\pi}x}$

#### 实现过程
- **使用欧拉方法求解贝塞尔方程：** 欧拉方法是求解常微分方程的简单数值方法，给定初始条件后，通过逐步积分得到数值解。欧拉方法的公式为：$$y_{n+1}=y_n+hf(x_n,y_n)$$
- **使用四阶龙格-库塔方法求解贝塞尔方程**：
	- 公式如下：$$\begin{gathered}
\text{7} k_1=hf(x_n,y_n) \\
k_{2} =hf\left(x_n+\frac h2,y_n+\frac{k_1}2\right) \\
\text{k} _{3}=hf\left(x_{n}+\frac{h}{2},y_{n}+\frac{k_{2}}{2}\right) \\
k_4=hf(x_n+h,y_n+k_3) \\
y_{n+1} =y_n+\frac{k_1+2k_2+2k_3+k_4}6 
\end{gathered}$$
- python编程求解
```python
import numpy as np
import matplotlib.pyplot as plt

# 定义贝塞尔方程的右端函数
def f(x, y, dy):
    return (1/x) * (n**2 - x**2) * y

# 初始条件
x0 = np.pi / 2
y0 = 2
dy0 = -2 / np.pi
h = 0.01
n = 0.5
x_end = 2 * np.pi

# 欧拉方法求解
x_euler = np.arange(x0, x_end, h)
y_euler = np.zeros(len(x_euler))
dy_euler = np.zeros(len(x_euler))
y_euler[0] = y0
dy_euler[0] = dy0

for i in range(1, len(x_euler)):
    y_euler[i] = y_euler[i-1] + h * dy_euler[i-1]
    dy_euler[i] = dy_euler[i-1] + h * f(x_euler[i-1], y_euler[i-1], dy_euler[i-1])

# 龙格-库塔方法求解
x_rk = np.arange(x0, x_end, h)
y_rk = np.zeros(len(x_rk))
dy_rk = np.zeros(len(x_rk))
y_rk[0] = y0
dy_rk[0] = dy0

for i in range(1, len(x_rk)):
    k1_y = h * dy_rk[i-1]
    k1_dy = h * f(x_rk[i-1], y_rk[i-1], dy_rk[i-1])
    
    k2_y = h * (dy_rk[i-1] + 0.5 * k1_dy)
    k2_dy = h * f(x_rk[i-1] + 0.5 * h, y_rk[i-1] + 0.5 * k1_y, dy_rk[i-1] + 0.5 * k1_dy)
    
    k3_y = h * (dy_rk[i-1] + 0.5 * k2_dy)
    k3_dy = h * f(x_rk[i-1] + 0.5 * h, y_rk[i-1] + 0.5 * k2_y, dy_rk[i-1] + 0.5 * k2_dy)
    
    k4_y = h * (dy_rk[i-1] + k3_dy)
    k4_dy = h * f(x_rk[i-1] + h, y_rk[i-1] + k3_y, dy_rk[i-1] + k3_dy)
    
    y_rk[i] = y_rk[i-1] + (k1_y + 2 * k2_y + 2 * k3_y + k4_y) / 6
    dy_rk[i] = dy_rk[i-1] + (k1_dy + 2 * k2_dy + 2 * k3_dy + k4_dy) / 6

# 精确解
x_exact = np.arange(x0, x_end, h)
y_exact = np.sin(x_exact) * np.sqrt(2 * np.pi / x_exact)

# 绘图
plt.figure(figsize=(12, 8))
plt.plot(x_euler, y_euler, label='Euler Method')
plt.plot(x_rk, y_rk, label='Runge-Kutta Method')
plt.plot(x_exact, y_exact, label='Exact Solution', linestyle='dashed')
plt.xlabel('x')
plt.ylabel('y')
plt.legend()
plt.title('Comparison of Numerical Methods for Solving Bessel Equation')
plt.show()

```
