#### 实现过程
- 参数设置（为了简化分析）
	- $L=0.5m$（质心到轮轴距离）
	- $R=0.1m$（车轮半径）
	- $m_l​=2kg$（左轮质量）
	- $m_r​=2kg$（右轮质量）
	- $J_{\theta} = 1 \, \text{kg} \cdot \text{m}^2$ （车体旋转惯量）
	- $J_{\phi} = 0.5 \, \text{kg} \cdot \text{m}^2$（车体转动惯量）
	- $J_l = 0.1 \, \text{kg} \cdot \text{m}^2$（左轮转动惯量）
	- $J_r = 0.1 \, \text{kg} \cdot \text{m}^2$（右轮转动惯量）
	- $mf = 0.3 \, \text{m}$（轮轴长度的一半）
	- $g = 9.81 \, \text{m/s}^2$（重力加速度）
- 编程求解
```python
import numpy as np
from scipy.integrate import solve_ivp

# 定义系统参数
L = 0.5  # 质心到轮轴距离 (m)
R = 0.1  # 车轮半径 (m)
m_l = 2  # 左轮质量 (kg)
m_r = 2  # 右轮质量 (kg)
J_theta = 1  # 车体旋转惯量 (kg*m^2)
J_phi = 0.5  # 车体转动惯量 (kg*m^2)
J_l = 0.1  # 左轮转动惯量 (kg*m^2)
J_r = 0.1  # 右轮转动惯量 (kg*m^2)
f = 0.3  # 轮轴长度的一半 (m)
g = 9.81  # 重力加速度 (m/s^2)

# 定义运动方程
def equations(t, y, m):
    theta, theta_dot = y
    # 动力学方程（简化版）
    theta_ddot = (m * g * L * np.sin(theta)) / (J_theta + m * L**2)
    return [theta_dot, theta_ddot]

# 平衡角度求解函数
def find_balance_angle(m, initial_angle):
    y0 = [initial_angle, 0]  # 初始角度和角速度
    t_span = [0, 10]  # 模拟时间
    sol = solve_ivp(equations, t_span, y0, args=(m,), method='RK45', t_eval=np.linspace(0, 10, 300))
    theta = sol.y[0]
    return np.max(np.abs(theta))

# 质量范围
m_values = np.linspace(30, 180, 10)
balance_angles = []

# 计算每个质量下的平衡角度范围
initial_angle = np.deg2rad(1)  # 初始角度偏差设置为1度
for m in m_values:
    balance_angle = find_balance_angle(m, initial_angle)
    balance_angles.append(balance_angle)
    print(f"质量: {m} kg, 平衡角度范围: {balance_angle} rad")

# 输出结果
print("质量与平衡角度范围:")
for m, angle in zip(m_values, balance_angles):
    print(f"质量: {m:.1f} kg, 平衡角度范围: {angle:.4f} rad")

```
