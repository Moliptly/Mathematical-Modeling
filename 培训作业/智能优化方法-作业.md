#### 作业分析
- 遗传算法步骤
	- **初始化**：生成初始种群
	- **选择**：根据适应度选择较优的个体
	- **交叉**：对选中的个体进行交叉操作，生成新的个体
	- **变异**：对新个体进行随机变异
	- **替换**：将新个体代替旧种群
	- **终止条件**：达到预设的代数或误差范围
- 算法参数设置
	- **种群大小**：100
	- **最大代数**：500
	- **交叉概率**：0.8
	- **变异概率**：0.1
	- **搜索范围**：$[-1, 2]$
- 代码实现
```python
import numpy as np
import matplotlib.pyplot as plt
# 目标函数
def obj(x):
	return x ** 2
	
# 遗传算法参数
pop_size = 100  # 种群大小
mut_rate = 0.1  # 变异率
num_gen = 1000  # 世代数
bounds = (-1, 2)  # 变量范围
precision = 0.0001  # 精度

# 初始化种群
def init_pop(size, bounds):
	return np.random.uniform(bounds[0], bounds[1], size)

# 计算适应度值
def comp_fit(pop):
	return obj(pop)

# 选择操作
def select(pop, fit):
	prob = fit / np.sum(fit)  # 根据适应度值计算选择概率
	indices = np.random.choice(np.arange(len(pop)), size=len(pop), p=prob)
	return pop[indices]

# 交叉操作
def cross(p1, p2):
	alpha = np.random.uniform(0, 1)  # 随机选择交叉比例
	return alpha * p1 + (1 - alpha) * p2  # 生成新个体

# 变异操作
def mutate(ind, mut_rate, bounds):
	if np.random.rand() < mut_rate: # 根据变异率决定是否变异
		ind += np.random.uniform(-0.1, 0.1)  # 对个体进行小范围变异
		ind = np.clip(ind, bounds[0], bounds[1])
	return ind

# 遗传算法主程序
def ga():
	pop = init_pop(pop_size, bounds) # 初始化种群
	best_fit_vals = []  # 用于存储每代的最佳适应度值
	for gen in range(num_gen):
		fit = comp_fit(pop)
		best_fit_vals.append(np.max(fit)) # 记录当前代的最佳适应度值
		sel_pop = select(pop, fit)  # 生成新的选择种群
		new_pop = []
		for i in range(0, pop_size, 2):
			p1, p2 = sel_pop[i], sel_pop[i+1]
			off1, off2 = cross(p1, p2), cross(p2, p1) # 交叉操作生成子代
			new_pop.append(mutate(off1, mut_rate, bounds)) # 子代变异并加入新种群
			new_pop.append(mutate(off2, mut_rate, bounds))
		pop = np.array(new_pop) # 更新种群
		if np.max(fit) >= (2 ** 2) - precision:
			break
	best_ind = pop[np.argmax(fit)] # 找到最佳个体
	return best_ind, np.max(fit), best_fit_vals

# 执行遗传算法
best_ind, best_fit, best_fit_vals = ga()
print("最佳个体:", best_ind)
print("最佳适应度值:", best_fit)
# 绘制收敛曲线
plt.plot(best_fit_vals)
plt.xlabel('Generation')
plt.ylabel('Fitness Value')
plt.title('Genetic Algorithm Optimization')
plt.show()
```
