#### 概念
误差：计算值与真实值之差，真值$x^*$，测量值$x$
绝对误差：$e = x^* - x$
相对误差：$e_{r}=\frac{e}{x^{*}}=\frac{x^{*}-x}{x^{*}}\approx\frac{x^{*}-x}{x}$
绝对误差限：$|e|=|x^{*}-x|\leq\varepsilon\quad\Longrightarrow x^{*}=x\pm\varepsilon$
灵敏度分析（误差的传播）：![[1719910679686.png]]
- 输入误差经过f(x)后，输出误差变为输入误差的$f^`(x)$ 倍
误差的类型：

- 舍入误差：四舍五入造成的
- 截断误差：方法误差，化简计算过程造成的

- $f(x)=cosx-x$
``` Mathematica
(* 定义函数和其导数 *)
f1[x_] := Cos[x] - x
f1Prime[x_] := -Sin[x] - 1

(* 初始区间和参数 *)
x0 = 0; x1 = 1; tol = 10^-6; maxIter = 100

(* 逐步搜索法 *)
stepSearch[f_, x0_, x1_, step_, tol_, maxIter_] := 
 Module[{x = x0, iter = 0, error = Abs[f[x]]},
  While[error > tol && iter < maxIter,
   x += step;
   error = Abs[f[x]];
   iter++;
   ];
  {x, iter}
  ]

(* 二分法 *)
bisectionMethod[f_, x0_, x1_, tol_, maxIter_] := 
 Module[{a = x0, b = x1, c, iter = 0, error},
  While[(b - a)/2 > tol && iter < maxIter,
   c = (a + b)/2;
   If[f[c] == 0, Break[]];
   If[Sign[f[c]] == Sign[f[a]], a = c, b = c];
   error = Abs[f[c]];
   iter++;
   ];
  {c, iter}
  ]

(* 牛顿法 *)
newtonMethod[f_, fPrime_, x0_, tol_, maxIter_] := 
 Module[{x = x0, xNew, iter = 0, error},
  error = Abs[f[x]];
  While[error > tol && iter < maxIter,
   xNew = x - f[x]/fPrime[x];
   error = Abs[xNew - x];
   x = xNew;
   iter++;
   ];
  {x, iter}
  ]

(* 弦截法 *)
secantMethod[f_, x0_, x1_, tol_, maxIter_] := 
 Module[{xPrev = x0, xCurr = x1, xNext, iter = 0, error},
  error = Abs[f[xCurr]];
  While[error > tol && iter < maxIter,
   xNext = xCurr - f[xCurr]*(xCurr - xPrev)/(f[xCurr] - f[xPrev]);
   error = Abs[xNext - xCurr];
   xPrev = xCurr;
   xCurr = xNext;
   iter++;
   ];
  {xNext, iter}
  ]

(* 比例求根法 *)
falsePositionMethod[f_, x0_, x1_, tol_, maxIter_] := 
 Module[{a = x0, b = x1, c, iter = 0, error},
  While[(b - a)/2 > tol && iter < maxIter,
   c = b - f[b]*(b - a)/(f[b] - f[a]);
   If[f[c] == 0, Break[]];
   If[Sign[f[c]] == Sign[f[a]], a = c, b = b, b = c];
   error = Abs[f[c]];
   iter++;
   ];
  {c, iter}
  ]

(* 求解和收敛速度对比 *)
step = 0.01;
rootStepSearch = stepSearch[f1, x0, x1, step, tol, maxIter];
rootBisection = bisectionMethod[f1, x0, x1, tol, maxIter];
rootNewton = newtonMethod[f1, f1Prime, x0, tol, maxIter];
rootSecant = secantMethod[f1, x0, x1, tol, maxIter];
rootFalsePosition = falsePositionMethod[f1, x0, x1, tol, maxIter];

(* 输出结果 *)
{rootStepSearch, rootBisection, rootNewton, rootSecant, rootFalsePosition}

(* 收敛速度对比图 *)
iterations = Range[0, maxIter];
errorsStepSearch = Table[Abs[f1[stepSearch[f1, x0, x1, step, tol, n][[1]]]], {n, iterations}];
errorsBisection = Table[Abs[f1[bisectionMethod[f1, x0, x1, tol, n][[1]]]], {n, iterations}];
errorsNewton = Table[Abs[f1[newtonMethod[f1, f1Prime, x0, tol, n][[1]]]], {n, iterations}];
errorsSecant = Table[Abs[f1[secantMethod[f1, x0, x1, tol, n][[1]]]], {n, iterations}];
errorsFalsePosition = Table[Abs[f1[falsePositionMethod[f1, x0, x1, tol, n][[1]]]], {n, iterations}];

(* 绘制收敛速度对比图 *)
ListLogPlot[{errorsStepSearch, errorsBisection, errorsNewton, errorsSecant, errorsFalsePosition}, 
 PlotLegends -> {"Step Search", "Bisection", "Newton", "Secant", "False Position"}, 
 PlotRange -> All, 
 AxesLabel -> {"Iterations", "Relative Error"}, 
 PlotLabel -> "Convergence Speed Comparison"]
```

``` matlab
% 定义函数和其导数
f1 = @(x) cos(x) - x;

% 逐步搜索法
% f: 要求根的函数
% x0: 初始猜测值
% x1: 结束猜测值（未使用）
% step: 每次迭代的步长
% tol: 容差，当 |f(x)| < tol 时停止迭代
% maxIter: 最大迭代次数
function [root, iter, errors] = stepSearch(f, x0, x1, step, tol, maxIter)
    % 初始化变量
    x = x0; % 从初始猜测值开始
    iter = 0; % 初始化迭代计数器
    errors = []; % 初始化误差数组，用于存储每次迭代的误差
    % 逐步搜索过程
    while abs(f(x)) > tol && iter < maxIter
        % 迭代条件：当函数值的绝对值大于容差并且迭代次数小于最大迭代次数时继续迭代
        x = x + step; % 更新 x 值，按步长 step 向前移动
        iter = iter + 1; % 增加迭代计数器
        errors(iter) = abs(f(x)); % 计算当前误差并存储
    end
    % 返回结果
    root = x; % 最终找到的根
end

% 二分法
% f: 要求根的函数
% x0: 初始区间的左端点
% x1: 初始区间的右端点
% tol: 容差，当 (b - a) / 2 < tol 时停止迭代
% maxIter: 最大迭代次数
function [root, iter, errors] = bisectionMethod(f, x0, x1, tol, maxIter)
    % 初始化变量
    a = x0; % 初始区间左端点
    b = x1; % 初始区间右端点
    iter = 0; % 初始化迭代计数器
    errors = []; % 初始化误差数组，用于存储每次迭代的误差
    % 二分法过程
    while (b - a) / 2 > tol && iter < maxIter
        % 迭代条件：当当前区间的一半长度大于容差并且迭代次数小于最大迭代次数时继续迭代
        c = (a + b) / 2; % 计算当前区间的中点
        if f(c) == 0
            % 如果中点的函数值为 0，则找到根，退出循环
            break;
        elseif sign(f(c)) == sign(f(a))
            % 如果中点的函数值与左端点的函数值符号相同，则将左端点移动到中点
            a = c;
        else
            % 否则，将右端点移动到中点
            b = c;
        end
        iter = iter + 1; % 增加迭代计数器
        errors(iter) = abs(f(c)); % 计算当前误差并存储
    end
    % 返回结果
    root = c; % 最终找到的根
end


% 牛顿法
% f: 要求根的函数
% f_prime: 函数 f 的导数
% x0: 初始猜测值
% tol: 容差，当 |f(x)| < tol 时停止迭代
% maxIter: 最大迭代次数
function [root, iter, errors] = newtonMethod(f, f_prime, x0, tol, maxIter)
    % 初始化变量
    x = x0; % 从初始猜测值开始
    iter = 0; % 初始化迭代计数器
    errors = []; % 初始化误差数组，用于存储每次迭代的误差
    % 牛顿法迭代过程
    while abs(f(x)) > tol && iter < maxIter
        % 迭代条件：当函数值的绝对值大于容差并且迭代次数小于最大迭代次数时继续迭代
        x = x - f(x) / f_prime(x); % 根据牛顿法更新 x 值
        iter = iter + 1; % 增加迭代计数器
        errors(iter) = abs(f(x)); % 计算当前误差并存储
    end
    % 返回结果
    root = x; % 最终找到的根
end


% 弦截法
% f: 要求根的函数
% x0: 初始猜测值 x_(k-1)
% x1: 初始猜测值 x_k
% tol: 容差，当 |f(x_curr)| < tol 时停止迭代
% maxIter: 最大迭代次数
function [root, iter, errors] = secantMethod(f, x0, x1, tol, maxIter)
    % 初始化变量
    x_prev = x0; % 初始猜测值 x_(k-1)
    x_curr = x1; % 初始猜测值 x_k
    iter = 0; % 初始化迭代计数器
    errors = []; % 初始化误差数组，用于存储每次迭代的误差
    % 弦截法迭代过程
    while abs(f(x_curr)) > tol && iter < maxIter
        % 迭代条件：当当前函数值的绝对值大于容差并且迭代次数小于最大迭代次数时继续迭代
        x_next = x_curr - f(x_curr) * (x_curr - x_prev) / (f(x_curr) - f(x_prev)); % 弦截法迭代公式
        x_prev = x_curr; % 更新 x_(k-1) 为当前值
        x_curr = x_next; % 更新 x_k 为下一次迭代值
        iter = iter + 1; % 增加迭代计数器
        errors(iter) = abs(f(x_curr)); % 计算当前误差并存储
    end
    % 返回结果
    root = x_curr; % 最终找到的根
end


% 比例求根法（假位法）
% f: 要求根的函数
% x0: 初始区间的左端点
% x1: 初始区间的右端点
% tol: 容差，当 (b - a) / 2 < tol 时停止迭代
% maxIter: 最大迭代次数
function [root, iter, errors] = falsePositionMethod(f, x0, x1, tol, maxIter)
    % 初始化变量
    a = x0; % 初始区间的左端点
    b = x1; % 初始区间的右端点
    iter = 0; % 初始化迭代计数器
    errors = []; % 初始化误差数组，用于存储每次迭代的误差
    % 比例求根法（假位法）迭代过程
    while (b - a) / 2 > tol && iter < maxIter
        % 迭代条件：当当前区间的一半长度大于容差并且迭代次数小于最大迭代次数时继续迭代
        c = b - f(b) * (b - a) / (f(b) - f(a)); % 计算当前区间的比例求根法中的新点 c
        if f(c) == 0
            % 如果新点 c 的函数值为 0，则找到根，退出循环
            break;
        elseif sign(f(c)) == sign(f(a))
            % 如果新点 c 的函数值与左端点 a 的函数值符号相同，则将左端点 a 移动到 c
            a = c;
        else
            % 否则，将右端点 b 移动到 c
            b = c;
        end
        iter = iter + 1; % 增加迭代计数器
        errors(iter) = abs(f(c)); % 计算当前误差并存储
    end
    % 返回结果
    root = c; % 最终找到的根
end
```
